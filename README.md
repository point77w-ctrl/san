const app = document.getElementById("app");
const $ = (id)=>document.getElementById(id);

const state = {
  subject: null, pool: [], exam: [], order: [], pos: 0, ans: [],
  graded: false, wrongOnly: false, showExp: false, score: null
};

function shuffle(arr){
  for(let i=arr.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[arr[i],arr[j]]=[arr[j],arr[i]]}
  return arr;
}

function setButtonsForNoExam(){
  $("btnGrade").disabled=true; $("btnPrev").disabled=true; $("btnNext").disabled=true;
  $("btnExp").disabled=true; $("btnWrong").disabled=true;
  $("btnGrade").innerText="채점"; $("btnWrong").innerText="오답만 모아보기"; $("btnExp").innerText="풀이";
}

function setButtonsForExam(){
  $("btnGrade").disabled = state.exam.length===0;
  $("btnPrev").disabled = state.exam.length<=1;
  $("btnNext").disabled = state.exam.length<=1;
  $("btnExp").disabled  = state.exam.length===0;
  $("btnWrong").disabled = true;
  $("btnGrade").innerText="채점"; $("btnWrong").innerText="오답만 모아보기"; $("btnExp").innerText="풀이";
}

function buildExam(){
  const usedQ = new Set();
  const base = shuffle([...state.pool]);
  const out = [];
  for(const it of base){
    if(out.length>=20) break;
    if(!it || !it.q || !it.o || it.o.length!==4) continue;
    if(usedQ.has(it.q)) continue;
    usedQ.add(it.q);
    out.push(it);
  }
  state.exam = out;
  state.order = out.map((_,i)=>i);
  state.pos = 0;
  state.ans = Array(out.length).fill(null);
  state.graded = false;
  state.wrongOnly = false;
  state.showExp = false;
  state.score = null;
  $("chipN").innerText = String(out.length);
  $("chipMode").innerText = "전체";
  setButtonsForExam();
}

function computeScore(){
  let correct=0, wrong=0, miss=0;
  for(let i=0;i<state.exam.length;i++){const a=state.ans[i]; if(a===null) miss++; else if(a===state.exam[i].a) correct++; else wrong++;}
  return {correct, wrong, miss, total: state.exam.length};
}

function esc(s){
  return String(s)
    .replace(/&/g,"&amp;")
    .replace(/</g,"&lt;")
    .replace(/>/g,"&gt;")
    .replace(/"/g,"&quot;")
    .replace(/'/g,"&#039;");
}

function renderSubjects(){
  const keys = Object.keys(BANK);
  let h = `<div class="panel"><div style="font-weight:900;margin-bottom:10px">과목 선택</div>`;
  keys.forEach((k,idx)=>{
    h += `<button class="subbtn" data-idx="${idx}">
            <b>${esc(k)}</b>
            <div class="small">문제은행: ${BANK[k].length}문항</div>
          </button>`;
  });
  h += `<div class="small" style="margin-top:10px">※ 과목 선택 후 “시험시작”</div></div>`;
  app.innerHTML = h;
}

function pickSubjectByIndex(idx){
  const keys = Object.keys(BANK);
  const name = keys[idx];
  if(!name) return;
  state.subject = name;
  $("chipSubj").innerText = name;
  app.innerHTML = `
    <div class="panel">
      <div style="font-weight:900;font-size:1.05rem;margin-bottom:6px">${esc(name)}</div>
      <div class="small">문제은행: ${BANK[name].length}문항</div>
      <div style="height:10px"></div>
      <button class="primary" id="btnStartInner">시험시작(랜덤 20)</button>
    </div>
  `;
}

function startExam(){
  if(!state.subject){ renderSubjects(); return; }
  state.pool = BANK[state.subject].map(x=>({q:x.q,o:x.o,a:x.a,exp:x.exp}));
  buildExam();
  renderQuestion();
}

function renderQuestion(){
  if(!state.exam.length){
    app.innerHTML = `<div class="panel"><b>문제가 없습니다.</b><div class="small">과목을 선택하고 시험시작을 누르세요.</div></div>`;
    return;
  }
  const qi = state.order[state.pos];
  const it = state.exam[qi];
  $("chipSubj").innerText = state.subject || "미선택";
  $("chipMode").innerText = state.wrongOnly ? "오답" : "전체";
  $("btnPrev").disabled = state.pos===0;
  $("btnNext").disabled = state.pos===state.order.length-1;

  let optsHTML = "";
  for(let oi=0; oi<4; oi++){ 
    const picked = state.ans[qi];
    let cls = "opt";
    if(picked===oi) cls += " sel";
    if(state.graded){
      if(oi===it.a) cls += " cor";
      else if(picked!==null && picked===oi && oi!==it.a) cls += " wrg";
    }
    optsHTML += `<div class="${cls}" data-opt="${oi}"><div class="opttxt">${esc(it.o[oi])}</div></div>`;
  }

  let scoreBox = "";
  if(state.score){
    scoreBox = `<div class="score">점수: ${state.score.correct}/${state.score.total}
      <span class="small"> (오답 ${state.score.wrong}, 미응답 ${state.score.miss})</span>
    </div>`;
  }

  app.innerHTML = `
    <div class="panel">
      <div class="qmeta">
        <div>${state.pos+1} / ${state.order.length}</div>
        <div class="small">${esc(state.subject||"")}</div>
      </div>
      <div class="qtext">${esc(it.q)}</div>
      ${optsHTML}
      ${scoreBox}
      <div class="exp" id="expBox">${esc(it.exp||"")}</div>
    </div>
  `;
  const expBox = $("expBox");
  if(expBox) expBox.style.display = state.showExp ? "block" : "none";
}

function choose(oi){
  if(state.graded) return;
  const qi = state.order[state.pos];
  state.ans[qi]=oi;
  renderQuestion();
}

function prev(){ if(state.pos>0){state.pos--; renderQuestion();} }
function next(){ if(state.pos<state.order.length-1){state.pos++; renderQuestion();} }

function toggleExplain(){ if(!state.exam.length) return; state.showExp=!state.showExp; renderQuestion(); }

function toggleGrade(){
  if(!state.exam.length) return;
  if(state.graded){
    state.graded=false; state.score=null; state.wrongOnly=false;
    state.order = state.exam.map((_,i)=>i); state.pos=0;
    $("btnGrade").innerText="채점";
    $("btnWrong").innerText="오답만 모아보기";
    $("btnWrong").disabled=true;
    $("chipMode").innerText="전체";
    renderQuestion();
    return;
  }
  state.graded=true;
  state.score=computeScore();
  const hasWrongOrMiss = (state.score.wrong + state.score.miss) > 0;
  $("btnWrong").disabled = !hasWrongOrMiss;
  $("btnGrade").innerText="채점취소";
  renderQuestion();
}

function toggleWrongOnly(){
  if(!state.graded) return;
  if(!state.wrongOnly){
    const wrongIdx = [];
    for(let i=0;i<state.exam.length;i++){ const a=state.ans[i]; if(a===null || a!==state.exam[i].a) wrongIdx.push(i); }
    state.order = wrongIdx.length ? wrongIdx : state.order;
    state.wrongOnly=true; state.pos=0;
    $("btnWrong").innerText="전체로 보기";
  } else {
    state.order = state.exam.map((_,i)=>i);
    state.wrongOnly=false; state.pos=0;
    $("btnWrong").innerText="오답만 모아보기";
  }
  renderQuestion();
}

function toggleTheme(){
  document.body.classList.toggle("light");
  const isLight = document.body.classList.contains("light");
  $("btnTheme").innerText = isLight ? "다크모드" : "라이트모드";
  localStorage.setItem("cbt_theme", isLight ? "light" : "dark");
}

function resetAll(){
  state.subject=null; state.pool=[]; state.exam=[]; state.order=[]; state.pos=0; state.ans=[];
  state.graded=false; state.wrongOnly=false; state.showExp=false; state.score=null;
  $("chipSubj").innerText="미선택"; $("chipN").innerText="0"; $("chipMode").innerText="전체";
  setButtonsForNoExam(); renderSubjects();
}

/* ===== 클릭/터치 이벤트 위임: 어떤 환경에서도 '안 눌림' 방지 ===== */
function handleAction(e){
  const t = e.target;
  const btn = t.closest("button");
  if(btn){
    const id = btn.id;
    if(id==="btnSubj") return renderSubjects();
    if(id==="btnStart") return startExam();
    if(id==="btnTheme") return toggleTheme();
    if(id==="btnReset") return resetAll();
    if(id==="btnWrong") return toggleWrongOnly();
    if(id==="btnGrade") return toggleGrade();
    if(id==="btnPrev") return prev();
    if(id==="btnNext") return next();
    if(id==="btnExp") return toggleExplain();
    if(id==="btnStartInner") return startExam();
    if(btn.classList.contains("subbtn")){
      const idx = parseInt(btn.getAttribute("data-idx"),10);
      return pickSubjectByIndex(idx);
    }
  }
  const opt = t.closest(".opt");
  if(opt){
    const oi = parseInt(opt.getAttribute("data-opt"),10);
    if(!Number.isNaN(oi)) return choose(oi);
  }
}
let __lastTouchTs = 0;
document.addEventListener("click", (e)=>{
  if(Date.now()-__lastTouchTs < 600) return;
  handleAction(e);
}, true);
document.addEventListener("touchend", (e)=>{
  __lastTouchTs = Date.now();
  if(e.cancelable) e.preventDefault();
  handleAction(e);
}, {capture:true, passive:false});
(function init(){
  const t = localStorage.getItem("cbt_theme");
  if(t==="light"){ document.body.classList.add("light"); $("btnTheme").innerText="다크모드"; }
  else { document.body.classList.remove("light"); $("btnTheme").innerText="라이트모드"; }
  $("chipSubj").innerText="미선택"; $("chipN").innerText="0"; $("chipMode").innerText="전체";
  setButtonsForNoExam();
  renderSubjects();
})();
</script>
